// Ler Numero Positivo
int leNumero(void)
{
    int num;
    do
    {
        printf("Insira o Numero:");
        scanf("%d", &num);
        if(num<=0)
        {
            printf("Numero Invalido!!!!\n\n");
        }
    }
    while(num<=0);

    return num;
}

// Area do Triangulo
float areaTriangulo(int base, int altura)
{
    float resultado = (float)(base * altura)/2;;
    return resultado;
}

// Area do Retangulo
float areaRetangulo(int c, int l)
{
    float resultado = c*l;
    return resultado;
}

// Area do Quadrado
float areaQuadrado(int l)
{
    float resultado = l*l;
    return resultado;
}

// Area do Circulo
float areaCirculo(int r)
{
    float resultado = PI * r * r;
    return resultado;
}
